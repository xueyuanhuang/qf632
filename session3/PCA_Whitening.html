import numpy as np
import matplotlib.pyplot as plt

np.random.seed(1)

mu = [3,3] 
sigma = [[7, 5],[5, 7]] # must be positive semi-definite
n = 1000
x = np.random.multivariate_normal(mu, sigma, size=n).T 

set1 = np.argsort(np.linalg.norm(x - 3, axis=0))[-20:] 
set2 = list(set(range(n)) - set(set1))

def plotting(x, xlim = 16, ylim = 16, title="Original"):
    fig, ax = plt.subplots()
    ax.scatter(x[0, set1], x[1, set1], s=20, c="red", alpha=0.2)
    ax.scatter(x[0, set2], x[1, set2], s=20, alpha=0.2)
    ax.set_aspect("equal")
    ax.set_xlim(-xlim, xlim)
    ax.set_ylim(-ylim, ylim)
    ax.set_xlabel("$x_1$")
    ax.set_ylabel("$x_2$")
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.set_title(title)

plt.rcParams['figure.figsize'] = [16, 16]
plotting(x,16,16,"Original")

    x[:,i] -= mu

#MAKE UNIT VARIANCE

for j in range(x.shape[0]):
    var = np.sum(np.square(x[j,:]), dtype = np.float32)/1000
    for i in range(1000):
        x[j,i] /=  np.sqrt(var)
    
plt.rcParams['figure.figsize'] = [16, 16]
plotting(x,10,10,"Unit Variance")

# WHITEN DATA

cov_mat = np.cov(x, rowvar=True)
Lambda, U = np.linalg.eigh(cov_mat)

y = U.T @ x #PCA applied
z = np.diag(Lambda**(-1/2)) @ y #data whitened

plt.rcParams['figure.figsize'] = [16, 16]
plotting(z,10,10,"Decorrelated/Orthogonalized/Whitened")
